## Task 2: CMake Configuration

**Files:**
- Create: `CMakeLists.txt`

**Step 1: Create root CMakeLists.txt**

```cmake
cmake_minimum_required(VERSION 3.24)
project(TablePro VERSION 0.1.0 LANGUAGES CXX)

# C++20 standard
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Qt Auto-MOC/UIC/RCC
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

# Build type
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Debug CACHE STRING "Build type" FORCE)
endif()

# Find Qt6
find_package(Qt6 6.6 REQUIRED COMPONENTS
    Core
    Gui
    Widgets
    Sql
    Network
    Concurrent
)

# Source files
set(TABLEPRO_SOURCES
    src/main.cpp
    src/ui/MainWindow.cpp
)

set(TABLEPRO_HEADERS
    src/ui/MainWindow.hpp
)

set(TABLEPRO_RESOURCES
    resources/resources.qrc
)

# Create executable
add_executable(tablepro
    ${TABLEPRO_SOURCES}
    ${TABLEPRO_HEADERS}
    ${TABLEPRO_RESOURCES}
)

# Link libraries
target_link_libraries(tablepro PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Widgets
    Qt6::Sql
    Qt6::Network
    Qt6::Concurrent
)

# Include directories
target_include_directories(tablepro PRIVATE
    ${CMAKE_SOURCE_DIR}/src
)

# Compiler warnings
target_compile_options(tablepro PRIVATE
    $<$<CXX_COMPILER_ID:MSVC>:/W4>
    $<$<OR:$<CXX_COMPILER_ID:GNU>,$<CXX_COMPILER_ID:Clang>,$<CXX_COMPILER_ID:AppleClang>>:-Wall -Wextra -Wpedantic>
)

# Enable testing
enable_testing()
```

## Task 6: MainWindow Implementation

**Files:**
- Create: `src/ui/MainWindow.cpp`

**Step 1: Create MainWindow.cpp**

```cpp
#include "MainWindow.hpp"
#include <QMenuBar>
#include <QToolBar>
#include <QStatusBar>
#include <QSplitter>
#include <QFile>

namespace tablepro {

MainWindow::MainWindow(QWidget* parent)
    : QMainWindow(parent)
    , m_mainSplitter(new QSplitter(Qt::Horizontal, this))
    , m_sidebarWidget(new QWidget(this))
    , m_mainContent(new QWidget(this))
{
    setWindowTitle(tr("TablePro"));
    setMinimumSize(1280, 720);
    resize(1400, 900);

    applyStyleSheet();
    setupMenuBar();
    setupToolBar();
    setupCentralWidget();
    setupStatusBar();
}

MainWindow::~MainWindow() = default;

void MainWindow::setupMenuBar() {
    // File menu
    auto* fileMenu = menuBar()->addMenu(tr("&File"));

    auto* newConnectionAction = fileMenu->addAction(tr("&New Connection..."));
    newConnectionAction->setShortcut(QKeySequence::New);

    auto* quitAction = fileMenu->addAction(tr("&Quit"));
    quitAction->setShortcut(QKeySequence::Quit);
    connect(quitAction, &QAction::triggered, qApp, &QApplication::quit);

    // Edit menu
    auto* editMenu = menuBar()->addMenu(tr("&Edit"));
    editMenu->addAction(tr("&Preferences..."));

    // View menu
    auto* viewMenu = menuBar()->addMenu(tr("&View"));
    viewMenu->addAction(tr("&Refresh"));
    viewMenu->addAction(tr("&Toggle Sidebar"));

    // Help menu
    auto* helpMenu = menuBar()->addMenu(tr("&Help"));
    helpMenu->addAction(tr("&About TablePro"));
}

void MainWindow::setupToolBar() {
    auto* toolbar = addToolBar(tr("Main Toolbar"));
    toolbar->setMovable(false);
    toolbar->setIconSize(QSize(24, 24));

    toolbar->addAction(tr("Connect"));
    toolbar->addSeparator();
    toolbar->addAction(tr("New Query"));
    toolbar->addAction(tr("Execute"));
}

void MainWindow::setupStatusBar() {
    statusBar()->showMessage(tr("Ready"));
}

void MainWindow::setupCentralWidget() {
    m_mainSplitter->addWidget(m_sidebarWidget);
    m_mainSplitter->addWidget(m_mainContent);
    m_mainSplitter->setSizes({250, 1150});

    setCentralWidget(m_mainSplitter);
}

void MainWindow::applyStyleSheet() {
    QFile styleFile(":/styles/dark.qss");
    if (styleFile.open(QIODevice::ReadOnly | QIODevice::Text)) {
        setStyleSheet(QString::fromUtf8(styleFile.readAll()));
    }
}

} // namespace tablepro
```

```makefile
.PHONY: all configure build clean run test format lint help

VERSION := $(shell git describe --tags --always --dirty 2>/dev/null || echo "dev")
BUILD_DIR := build/debug

all: build

configure:
	cmake --preset debug

build: configure
	cmake --build $(BUILD_DIR) -j$$(nproc 2>/dev/null || sysctl -n hw.ncpu)

clean:
	rm -rf build/

run: build
	./$(BUILD_DIR)/tablepro

test: build
	cd $(BUILD_DIR) && ctest --output-on-failure

format:
	find src -name "*.cpp" -o -name "*.hpp" | xargs clang-format -i

lint:
	clang-tidy src/**/*.cpp -- -I$(BUILD_DIR)

release:
	cmake --preset release
	cmake --build build/release -j$$(nproc 2>/dev/null || sysctl -n hw.ncpu)

	@echo ""

	@echo "  make build       - Build the project"
```

## Task 10: GitHub Actions CI

**Files:**
- Create: `.github/workflows/ci.yml`

**Step 1: Create ci.yml**

```yaml
name: CI

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  build-macos:
    runs-on: macos-14
    steps:
      - uses: actions/checkout@v4

      - name: Install dependencies
        run: |
          brew install qt@6 cmake ninja
          echo "Qt6_DIR=$(brew --prefix qt@6)/lib/cmake/Qt6" >> $GITHUB_ENV

      - name: Configure
        run: cmake -B build -G Ninja -DCMAKE_BUILD_TYPE=Debug

      - name: Build
        run: cmake --build build -j$(sysctl -n hw.ncpu)

      - name: Test
        run: cd build && ctest --output-on-failure

  build-linux:
    runs-on: ubuntu-22.04
    steps:
      - uses: actions/checkout@v4

      - name: Install dependencies
        run: |
          sudo apt-get update
          sudo apt-get install -y qt6-base-dev qt6-tools-dev-tools ninja-build

      - name: Configure
        run: cmake -B build -G Ninja -DCMAKE_BUILD_TYPE=Debug

      - name: Build
        run: cmake --build build -j$(nproc)

      - name: Test
        run: cd build && ctest --output-on-failure

  build-windows:
    runs-on: windows-2022
    steps:
      - uses: actions/checkout@v4

      - name: Install Qt
        uses: jurplel/install-qt-action@v3
        with:
          version: '6.6.1'
          arch: 'win64_msvc2019_64'

      - name: Configure
        run: cmake -B build -G "Visual Studio 17 2022" -A x64

      - name: Build
        run: cmake --build build --config Debug

      - name: Test
        run: cd build && ctest -C Debug --output-on-failure
```
